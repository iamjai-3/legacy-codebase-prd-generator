```src/schemas/subchapter.schema.ts
import {{ z }} from 'zod';

export const subchapterCreateSchema = z.object({{
  fleet: z.string().min(1, 'Fleet is required').max(10),
  chapter: z.string().min(1, 'Chapter is required').max(3),
  subchapter: z.string().min(1, 'Subchapter is required').max(2),
  description: z.string().max(50).optional(),
  exclusionFlag: z.boolean().optional(),
  etops: z.boolean().optional(),
  alertLimitLandings: z.number().min(0).optional(),
  alertLimitDays: z.number().min(0).optional(),
}});

export const subchapterUpdateSchema = z.object({{
  subchaptersId: z.number(),
  description: z.string().max(50).optional(),
  exclusionFlag: z.boolean().optional(),
  etops: z.boolean().optional(),
  alertLimitLandings: z.number().min(0).optional(),
  alertLimitDays: z.number().min(0).optional(),
}});

export type SubchapterCreateInput = z.infer<typeof subchapterCreateSchema>;
export type SubchapterUpdateInput = z.infer<typeof subchapterUpdateSchema>;
```

```src/hooks/api/useSubchapters.ts
import {{ useQuery, useMutation, useQueryClient }} from '@tanstack/react-query';
import {{ api }} from '@/lib/api';
import type {{ Subchapter, PagedResult }} from '@/types/{module_name}';
import type {{ SubchapterCreateInput, SubchapterUpdateInput }} from '@/schemas/subchapter.schema';

const QUERY_KEY = 'subchapters';

export function usePagedSubchapters(pageNumber: number, pageSize: number) {{
  return useQuery({{
    queryKey: [QUERY_KEY, 'paged', pageNumber, pageSize],
    queryFn: async () => {{
      const {{ data }} = await api.get<PagedResult<Subchapter>>(
        `/api/subchapters/paged?pageNumber=${{pageNumber}}&pageSize=${{pageSize}}`
      );
      return data;
    }},
  }});
}}

export function useSubchapter(id: number, enabled: boolean = true) {{
  return useQuery({{
    queryKey: [QUERY_KEY, id],
    queryFn: async () => {{
      const {{ data }} = await api.get<Subchapter>(`/api/subchapters/${{id}}`);
      return data;
    }},
    enabled,
  }});
}}

export function useCreateSubchapter() {{
  const queryClient = useQueryClient();
  return useMutation({{
    mutationFn: async (input: SubchapterCreateInput) => {{
      const {{ data }} = await api.post<Subchapter>('/api/subchapters', input);
      return data;
    }},
    onSuccess: () => {{
      queryClient.invalidateQueries({{ queryKey: [QUERY_KEY] }});
    }},
  }});
}}

export function useUpdateSubchapter() {{
  const queryClient = useQueryClient();
  return useMutation({{
    mutationFn: async (input: SubchapterUpdateInput) => {{
      const {{ data }} = await api.put<Subchapter>(
        `/api/subchapters/${{input.subchaptersId}}`,
        input
      );
      return data;
    }},
    onSuccess: () => {{
      queryClient.invalidateQueries({{ queryKey: [QUERY_KEY] }});
    }},
  }});
}}

export function useDeleteSubchapter() {{
  const queryClient = useQueryClient();
  return useMutation({{
    mutationFn: async (id: number) => {{
      await api.delete(`/api/subchapters/${{id}}`);
    }},
    onSuccess: () => {{
      queryClient.invalidateQueries({{ queryKey: [QUERY_KEY] }});
    }},
  }});
}}
```

```src/components/{module_name}/subchapters/SubchapterForm.tsx
import {{ useEffect }} from 'react';
import {{ useNavigate, useParams }} from 'react-router-dom';
import {{ useForm }} from 'react-hook-form';
import {{ zodResolver }} from '@hookform/resolvers/zod';
import {{ useSubchapter, useCreateSubchapter, useUpdateSubchapter }} from '@/hooks/api/useSubchapters';
import {{ subchapterCreateSchema, subchapterUpdateSchema, type SubchapterCreateInput, type SubchapterUpdateInput }} from '@/schemas/subchapter.schema';
import {{ Button }} from '@/components/ui/button';
import {{ Card, CardContent, CardDescription, CardHeader, CardTitle }} from '@/components/ui/card';
import {{ Form, FormControl, FormDescription, FormField, FormItem, FormLabel, FormMessage }} from '@/components/ui/form';
import {{ Input }} from '@/components/ui/input';
import {{ Checkbox }} from '@/components/ui/checkbox';
import {{ ArrowLeft, Loader2, Save }} from 'lucide-react';
import {{ toast }} from 'sonner';

export function SubchapterForm() {{
  const navigate = useNavigate();
  const {{ id }} = useParams<{{ id: string }}>();
  const isEditMode = !!id;
  const subchapterId = id ? Number(id) : undefined;

  const {{ data: subchapter, isLoading: loadingSubchapter }} = useSubchapter(subchapterId || 0, isEditMode);
  const createMutation = useCreateSubchapter();
  const updateMutation = useUpdateSubchapter();

  const form = useForm<SubchapterCreateInput | SubchapterUpdateInput>({{
    resolver: zodResolver(isEditMode ? subchapterUpdateSchema : subchapterCreateSchema),
    defaultValues: isEditMode
      ? {{
          subchaptersId: subchapterId,
          description: '',
          exclusionFlag: false,
          etops: false,
        }}
      : {{
          fleet: '',
          chapter: '',
          subchapter: '',
          description: '',
          exclusionFlag: false,
          etops: false,
        }},
  }});

  useEffect(() => {{
    if (subchapter && isEditMode) {{
      form.reset({{
        subchaptersId: subchapter.subchaptersId,
        description: subchapter.description || '',
        exclusionFlag: subchapter.exclusionFlag ?? false,
        etops: subchapter.etops ?? false,
        alertLimitLandings: subchapter.alertLimitLandings ?? undefined,
        alertLimitDays: subchapter.alertLimitDays ?? undefined,
      }});
    }}
  }}, [subchapter, isEditMode, form]);

  const onSubmit = async (values: SubchapterCreateInput | SubchapterUpdateInput) => {{
    try {{
      if (isEditMode) {{
        await updateMutation.mutateAsync(values as SubchapterUpdateInput);
        toast.success('Subchapter updated successfully');
      }} else {{
        await createMutation.mutateAsync(values as SubchapterCreateInput);
        toast.success('Subchapter created successfully');
      }}
      navigate(-1);
    }} catch (error: any) {{
      toast.error(error?.message || 'An error occurred');
    }}
  }};

  const isSubmitting = createMutation.isPending || updateMutation.isPending;

  if (loadingSubchapter && isEditMode) {{
    return (
      <div className="flex justify-center items-center p-8">
        <Loader2 className="h-8 w-8 animate-spin text-muted-foreground" />
      </div>
    );
  }}

  return (
    <div className="space-y-6">
      <div className="flex items-center gap-4">
        <Button variant="ghost" size="sm" onClick={{() => navigate(-1)}}>
          <ArrowLeft className="mr-2 h-4 w-4" />
          Back
        </Button>
        <div>
          <h1 className="text-3xl font-bold tracking-tight">
            {{isEditMode ? 'Edit Subchapter' : 'Create Subchapter'}}
          </h1>
        </div>
      </div>

      <Card>
        <CardHeader>
          <CardTitle>Subchapter Information</CardTitle>
          <CardDescription>Enter the subchapter details below.</CardDescription>
        </CardHeader>
        <CardContent>
          <Form {{...form}}>
            <form onSubmit={{form.handleSubmit(onSubmit)}} className="space-y-6">
              {{!isEditMode && (
                <div className="grid grid-cols-3 gap-4">
                  <FormField
                    control={{form.control}}
                    name="fleet"
                    render={{({{ field }}) => (
                      <FormItem>
                        <FormLabel>Fleet *</FormLabel>
                        <FormControl>
                          <Input {{...field}} maxLength={{10}} className="uppercase" />
                        </FormControl>
                        <FormMessage />
                      </FormItem>
                    )}}
                  />
                  <FormField
                    control={{form.control}}
                    name="chapter"
                    render={{({{ field }}) => (
                      <FormItem>
                        <FormLabel>Chapter *</FormLabel>
                        <FormControl>
                          <Input {{...field}} maxLength={{3}} className="uppercase" />
                        </FormControl>
                        <FormMessage />
                      </FormItem>
                    )}}
                  />
                  <FormField
                    control={{form.control}}
                    name="subchapter"
                    render={{({{ field }}) => (
                      <FormItem>
                        <FormLabel>Subchapter *</FormLabel>
                        <FormControl>
                          <Input {{...field}} maxLength={{2}} className="uppercase" />
                        </FormControl>
                        <FormMessage />
                      </FormItem>
                    )}}
                  />
                </div>
              )}}

              <FormField
                control={{form.control}}
                name="description"
                render={{({{ field }}) => (
                  <FormItem>
                    <FormLabel>Description</FormLabel>
                    <FormControl>
                      <Input {{...field}} maxLength={{50}} />
                    </FormControl>
                    <FormMessage />
                  </FormItem>
                )}}
              />

              <div className="flex gap-4 pt-4">
                <Button type="submit" disabled={{isSubmitting}}>
                  {{isSubmitting ? (
                    <><Loader2 className="mr-2 h-4 w-4 animate-spin" />{{isEditMode ? 'Updating...' : 'Creating...'}}</>
                  ) : (
                    <><Save className="mr-2 h-4 w-4" />{{isEditMode ? 'Update' : 'Create'}}</>
                  )}}
                </Button>
                <Button type="button" variant="outline" onClick={{() => navigate(-1)}} disabled={{isSubmitting}}>
                  Cancel
                </Button>
              </div>
            </form>
          </Form>
        </CardContent>
      </Card>
    </div>
  );
}}
```

```src/components/{module_name}/subchapters/SubchapterList.tsx
import {{ useState }} from 'react';
import {{ useNavigate }} from 'react-router-dom';
import {{ usePagedSubchapters, useDeleteSubchapter }} from '@/hooks/api/useSubchapters';
import {{ Button }} from '@/components/ui/button';
import {{ Table, TableBody, TableCell, TableHead, TableHeader, TableRow }} from '@/components/ui/table';
import {{ AlertDialog, AlertDialogAction, AlertDialogCancel, AlertDialogContent, AlertDialogDescription, AlertDialogFooter, AlertDialogHeader, AlertDialogTitle }} from '@/components/ui/alert-dialog';
import {{ Badge }} from '@/components/ui/badge';
import {{ Card, CardContent }} from '@/components/ui/card';
import {{ Loader2, Plus, Edit, Trash2, Eye }} from 'lucide-react';
import {{ toast }} from 'sonner';
import type {{ Subchapter }} from '@/types/{module_name}';

export function SubchapterList() {{
  const navigate = useNavigate();
  const [page, setPage] = useState(1);
  const pageSize = 20;
  
  const {{ data, isLoading, error }} = usePagedSubchapters(page, pageSize);
  const deleteMutation = useDeleteSubchapter();

  const [deleteDialogOpen, setDeleteDialogOpen] = useState(false);
  const [itemToDelete, setItemToDelete] = useState<Subchapter | null>(null);

  const handleDeleteClick = (item: Subchapter) => {{
    setItemToDelete(item);
    setDeleteDialogOpen(true);
  }};

  const handleDeleteConfirm = async () => {{
    if (!itemToDelete) return;
    try {{
      await deleteMutation.mutateAsync(itemToDelete.subchaptersId);
      toast.success('Subchapter deleted successfully');
      setDeleteDialogOpen(false);
      setItemToDelete(null);
    }} catch (error: any) {{
      toast.error(error.message || 'Failed to delete');
    }}
  }};

  if (error) {{
    return (
      <Card>
        <CardContent className="p-8">
          <p className="text-destructive">Failed to load data: {{error.message}}</p>
        </CardContent>
      </Card>
    );
  }}

  return (
    <div className="space-y-4">
      <div className="flex justify-between items-center">
        <div>
          <h1 className="text-3xl font-bold tracking-tight">Subchapters</h1>
          <p className="text-muted-foreground">Manage subchapter records</p>
        </div>
        <Button onClick={{() => navigate('/subchapters/new')}}>
          <Plus className="mr-2 h-4 w-4" />
          Create Subchapter
        </Button>
      </div>

      <Card>
        <CardContent className="p-0">
          {{isLoading ? (
            <div className="flex justify-center items-center p-8">
              <Loader2 className="h-8 w-8 animate-spin text-muted-foreground" />
            </div>
          ) : (
            <>
              <Table>
                <TableHeader>
                  <TableRow>
                    <TableHead>Fleet</TableHead>
                    <TableHead>Chapter</TableHead>
                    <TableHead>Subchapter</TableHead>
                    <TableHead>Description</TableHead>
                    <TableHead className="text-right">Actions</TableHead>
                  </TableRow>
                </TableHeader>
                <TableBody>
                  {{data?.items.length === 0 ? (
                    <TableRow>
                      <TableCell colSpan={{5}} className="text-center py-8 text-muted-foreground">
                        No records found.
                      </TableCell>
                    </TableRow>
                  ) : (
                    data?.items.map((item) => (
                      <TableRow key={{item.subchaptersId}}>
                        <TableCell className="font-mono font-semibold">{{item.fleet}}</TableCell>
                        <TableCell className="font-mono font-semibold">{{item.chapter}}</TableCell>
                        <TableCell className="font-mono font-semibold">{{item.subchapter}}</TableCell>
                        <TableCell>{{item.description || '-'}}</TableCell>
                        <TableCell className="text-right">
                          <div className="flex justify-end gap-2">
                            <Button variant="ghost" size="sm" onClick={{() => navigate(`/subchapters/${{item.subchaptersId}}`)}}>  
                              <Eye className="h-4 w-4" />
                            </Button>
                            <Button variant="ghost" size="sm" onClick={{() => navigate(`/subchapters/${{item.subchaptersId}}/edit`)}}>
                              <Edit className="h-4 w-4" />
                            </Button>
                            <Button variant="ghost" size="sm" onClick={{() => handleDeleteClick(item)}} disabled={{deleteMutation.isPending}}>
                              <Trash2 className="h-4 w-4 text-destructive" />
                            </Button>
                          </div>
                        </TableCell>
                      </TableRow>
                    ))
                  )}}
                </TableBody>
              </Table>

              {{data && data.totalPages > 1 && (
                <div className="flex items-center justify-between border-t px-4 py-4">
                  <div className="text-sm text-muted-foreground">
                    Page {{data.pageNumber}} of {{data.totalPages}} ({{data.totalCount}} total)
                  </div>
                  <div className="flex gap-2">
                    <Button variant="outline" size="sm" onClick={{() => setPage(page - 1)}} disabled={{!data.hasPreviousPage || isLoading}}>
                      Previous
                    </Button>
                    <Button variant="outline" size="sm" onClick={{() => setPage(page + 1)}} disabled={{!data.hasNextPage || isLoading}}>
                      Next
                    </Button>
                  </div>
                </div>
              )}}
            </>
          )}}
        </CardContent>
      </Card>

      <AlertDialog open={{deleteDialogOpen}} onOpenChange={{setDeleteDialogOpen}}>
        <AlertDialogContent>
          <AlertDialogHeader>
            <AlertDialogTitle>Are you sure?</AlertDialogTitle>
            <AlertDialogDescription>
              This will permanently delete this record. This action cannot be undone.
            </AlertDialogDescription>
          </AlertDialogHeader>
          <AlertDialogFooter>
            <AlertDialogCancel>Cancel</AlertDialogCancel>
            <AlertDialogAction onClick={{handleDeleteConfirm}} className="bg-destructive text-destructive-foreground hover:bg-destructive/90">
              {{deleteMutation.isPending ? (
                <><Loader2 className="mr-2 h-4 w-4 animate-spin" />Deleting...</>
              ) : (
                'Delete'
              )}}
            </AlertDialogAction>
          </AlertDialogFooter>
        </AlertDialogContent>
      </AlertDialog>
    </div>
  );
}}
```
